                                                if (stacks->top1 + 1 < stacks->top2) {
                                                    stacks->arr[++stacks->top1] = value;
                                                    return 1;
                                                }
                                                return 0;
                                            }
                                            
                                            int pushStack2(struct TwoStacks* stacks, int value) {

                                                    stacks->arr[--stacks->top2] = value;
                                                    return 1;
                                                }
                                                return 0;
                                            }

                                                #include < stdio.h>

                                                #define MAX 100
                                                
                                                struct Stack {
                                                    int arr[MAX];
                                                    int top;
                                                };
                                                
                                                void createStack(struct Stack* stack) {
                                                    stack->top = -1;
                                                }
                                                
                                                int isFull(struct Stack* stack) {
                                                    return stack->top == MAX - 1;
                                                }
                                                
                                                int isEmpty(struct Stack* stack) {
                                                    return stack->top == -1;
                                                }
                                                
                                                void push(struct Stack* stack, int value) {
                                                    if (!isFull(stack)) {
                                                        stack->arr[++stack->top] = value;
                                                    } else {
                                                        printf("Stack Overflow\n");
                                                    }
                                                }
                                                
                                                int pop(struct Stack* stack) {
                                                    if (!isEmpty(stack)) {
                                                        return stack->arr[stack->top--];
                                                    } else {
                                                        printf("Stack Underflow\n");
                                                        return -1; // Return -1 for underflow
                                                    }
                                                }
                                                
                                                int peek(struct Stack* stack) {
                                                    if (!isEmpty(stack)) {
                                                        return stack->arr[stack->top];
                                                    }
                                                    return -1; // Return -1 if empty
                                                }
                                                
                                                void sortStack(struct Stack* stack) {
                                                    int i, j, temp;
                                                    for (i = 0; i <= stack->top; i++) {
                                                        for (j = 0; j < stack->top - i; j++) {
                                                            if (stack->arr[j] > stack->arr[j + 1]) {
                                                                temp = stack->arr[j];
                                                                stack->arr[j] = stack->arr[j + 1];
                                                                stack->arr[j + 1] = temp;
                                                            }
                                                        }
                                                    }
                                                }
                                                
                                                void displayStack(struct Stack* stack) {
                                                    for (int i = stack->top; i >= 0; i--) {
                                                        printf("%d\n", stack->arr[i]);
                                                    }
                                                }
                                                
                                                int main() {
                                                    struct Stack stack;
                                                    createStack(&stack);
                                                
                                                    int n, value;
                                                
                                                    printf("Enter the number of elements to push onto the stack: ");
                                                    scanf("%d", &n);
                                                
                                                    for (int i = 0; i < n; i++) {
                                                        printf("Enter value %d: ", i + 1);
                                                        scanf("%d", &value);
                                                        push(&stack, value);
                                                    }
                                                
                                                    printf("Original Stack:\n");
                                                    displayStack(&stack);
                                                
                                                    sortStack(&stack);
                                                
                                                    printf("Sorted Stack in Ascending Order:\n");
                                                    displayStack(&stack);
                                                
                                                    return 0;
                                                }

                                                    #include < stdio.h>

                                                    #define MAX 100
                                                    
                                                    struct Stack {
                                                        char arr[MAX];
                                                        int top;
                                                    };
                                                    
                                                    void createStack(struct Stack* stack) {
                                                        stack->top = -1;
                                                    }
                                                    
                                                    int isFull(struct Stack* stack) {
                                                        return stack->top == MAX - 1;
                                                    }
                                                    
                                                    int isEmpty(struct Stack* stack) {
                                                        return stack->top == -1;
                                                    }
                                                    
                                                    void push(struct Stack* stack, char value) {
                                                        if (!isFull(stack)) {
                                                            stack->arr[++stack->top] = value;
                                                        }
                                                    }
                                                    
                                                    char pop(struct Stack* stack) {
                                                        if (!isEmpty(stack)) {
                                                            return stack->arr[stack->top--];
                                                        }
                                                        return '\0';
                                                    }
                                                    
                                                    char peek(struct Stack* stack) {
                                                        if (!isEmpty(stack)) {
                                                            return stack->arr[stack->top];
                                                        }
                                                        return '\0';
                                                    }
                                                    
                                                    int precedence(char op) {
                                                        if (op == '+' || op == '-') return 1;
                                                        if (op == '*' || op == '/') return 2;
                                                        if (op == '^') return 3;
                                                        return 0;
                                                    }
                                                    
                                                    void reverse(char* exp) {
                                                        int n = 0;
                                                        while (exp[n] != '\0') n++;
                                                        for (int i = 0; i < n / 2; i++) {
                                                            char temp = exp[i];
                                                            exp[i] = exp[n - i - 1];
                                                            exp[n - i - 1] = temp;
                                                        }
                                                    }
                                                    
                                                    void infixToPrefix(char* infix, char* prefix) {
                                                        struct Stack stack;
                                                        createStack(&stack);
                                                        reverse(infix);
                                                        
                                                        int j = 0;
                                                        for (int i = 0; infix[i] != '\0'; i++) {
                                                            char current = infix[i];
                                                    
                                                            if ((current >= 'A' && current <= 'Z') || (current >= 'a' && current <= 'z')) {
                                                                prefix[j++] = current;
                                                            } else if (current == ')') {
                                                                push(&stack, current);
                                                            } else if (current == '(') {
                                                                while (!isEmpty(&stack) && peek(&stack) != ')') {
                                                                    prefix[j++] = pop(&stack);
                                                                }
                                                                pop(&stack);
                                                            } else {
                                                                while (!isEmpty(&stack) && precedence(peek(&stack)) >= precedence(current)) {
                                                                    prefix[j++] = pop(&stack);
                                                                }
                                                                push(&stack, current);
                                                            }
                                                        }
