                    int main() {
                        char str[100]; 
    
                        printf("Enter a string: ");
                        gets(str); // Read a line of input until newline
                    
                        // Convert the string to uppercase
                        for (int i = 0; str[i] != '\0'; i++) {
                            // Check if the character is a lowercase letter
                            if (str[i] >= 'a' && str[i] <= 'z') {
                                // Convert to uppercase by subtracting 32 from its ASCII value
                                str[i] = str[i] - 32;
                            }
                        }
                    
                        // Print the converted string
                        printf("Uppercase string: %s\n", str);
                    
                        return 0;
                    }

    if (left == right) {
    
        for (int i = 0; str[i] != '\0'; i++) {
            printf("%c", str[i]);

        }
    }
}

int countOccurrences(char *sentence, char *word) {
    int count = 0;
    char *p = sentence;

    while (*p) {
        
        char *s = p;
        char *w = word;

        while (*s == *w && *w != '\0') {
            s++;
            w++;
        }

    
        if (*w == '\0') {
            
            if ((s == sentence || *(s) == ' ') && (p == sentence || *(p - 1) == ' ')) {
                count++;
            }
        }
        p++; 
    }

    return count;
}

void replaceWord(char *sentence, char *oldWord, char *newWord) {
    char result[1000];
    int i = 0, j = 0, k;

    while (sentence[i] != '\0') {
        
        k = 0;
        int match = 1; 

        while (oldWord[k] != '\0') {
            if (sentence[i + k] != oldWord[k]) {
                match = 0; 
                break;
            }
            k++;
        }

        
        if (match && (sentence[i + k] == ' ' || sentence[i + k] == '\0')) {
            for (k = 0; newWord[k] != '\0'; k++) {
                result[j++] = newWord[k]; 
            }
            
            i += k;
        } else {
            result[j++] = sentence[i]; 
            i++;
        }
    }
    
    result[j] = '\0'; 

    
    printf("Modified sentence: %s\n", result);
}

void trimSpaces(char *str) {
    int start = 0, end = 0;

    
    while (str[start] == ' ') {
        start++;
    }

    
    end = start;

        end++;
    }
    end--; 
    while (end >= start && str[end] == ' ') {
        end--;

    int j = 0;

  //copy
    for (int i = start; i <= end; i++) {

        str[i] = trimmed[i];

}

void deleteWord(char words[MAX_WORDS][MAX_LENGTH], char meanings[MAX_WORDS][MAX_LENGTH], char *wordToDelete) {
    int i, found = 0;

    for (i = 0; i < MAX_WORDS; i++) {
        if (compareStrings(words[i], wordToDelete) == 0) {
            found = 1;
            
            for (int j = i; j < MAX_WORDS - 1; j++) {
                copyString(words[j], words[j + 1]);
                copyString(meanings[j], meanings[j + 1]);
            }
           
            copyString(words[MAX_WORDS - 1], "");
            copyString(meanings[MAX_WORDS - 1], "");
            printf("Deleted word: %s\n", wordToDelete);
            break;
        }
    }

    if (!found) {
        printf("Word not found: %s\n", wordToDelete);
    }
}


void searchWord(char words[MAX_WORDS][MAX_LENGTH], char meanings[MAX_WORDS][MAX_LENGTH], char *wordToSearch) {
    int i, found = 0;

    for (i = 0; i < MAX_WORDS; i++) {
        if (compareStrings(words[i], wordToSearch) == 0) {
            found = 1;
            printf("Meaning of %s: %s\n", words[i], meanings[i]);

        }
    }

    }
}


void copyString(char *dest, char *src) {
    while (*src != '\0') {
        *dest = *src;
        dest++;
        src++;
    }
    *dest = '\0';
}


int compareStrings(char *str1, char *str2) {
    while (*str1 != '\0' && *str2 != '\0') {
        if (*str1 != *str2) {
            return 1;
        }
        str1++;
        str2++;
    }
    return (*str1 == *str2) ? 0 : 1; 
}
