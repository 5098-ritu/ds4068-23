Recursion Program:

  • 1. Calculate factorial of a number using recursion
  • 2. Calculate fibonacci of a number using recursion
  • 3. Calculate the sum of natural numbers till 10 using recursion
  • 4. Convert a number into binary using recursion
  • 5. Find the smallest element in the integer array using recursion
  • 6. Enter the number from the user and add all digits of the number using recursion
  • 7. Find the arithmetic mean of an integer array using recursion

Code 1


                    #include < stdio.h>

                    long long factorial(int n) {
                        if (n == 0)
                            return 1;
                        else
                            return n * factorial(n - 1);
                    }
                    
                    int main() {
                        int number;
                        
                        printf("Enter a positive integer: ");
                        scanf("%d", &number);
                        
                        if (number < 0) {
                            printf("Factorial is not defined for negative numbers.\n");
                        } else {
                            printf("Factorial of %d = %lld\n", number, factorial(number));
                        }
                    
                        return 0;
                    }

Output


                    Enter a positive integer: 5
                    Factorial of 5 = 120
        
                    Enter a positive integer: -3
                    Factorial is not defined for negative numbers.

Code 2


                    #include < stdio.h>

                    int fibonacci(int n) {
                        if (n <= 1)
                            return n;
                        return fibonacci(n - 1) + fibonacci(n - 2);
                    }
                    
                    int main() {
                        int number;
                        
                        printf("Enter a positive integer: ");
                        scanf("%d", &number);
                        
                        if (number < 0) {
                            printf("Fibonacci is not defined for negative numbers.\n");
                        } else {
                            printf("Fibonacci of %d = %d\n", number, fibonacci(number));
                        }
                    
                        return 0;
                    }

Output


                    Enter a positive integer: 5
                    Fibonacci of 5 = 5
        
                    Enter a positive integer: -2
                    Fibonacci is not defined for negative numbers.

Code 3


                    #include < stdio.h>

                    int sum(int n) {
                        if (n == 0)
                            return 0;
                        return n + sum(n - 1);
                    }
                    
                    int main() {
                        int number = 10;
                        int result = sum(number);
                        
                        printf("Sum of natural numbers till %d = %d\n", number, result);
                        
                        return 0;
                    }

Output


                    Sum of natural numbers till 10 = 55

Code 4


                    #include < stdio.h>

                    void binary(int n) {
                        if (n > 1)
                            binary(n / 2);
                        printf("%d", n % 2);
                    }
                    
                    int main() {
                        int number;
                    
                        printf("Enter a positive integer: ");
                        scanf("%d", &number);
                    
                        if (number < 0) {
                            printf("Please enter a non-negative integer.\n");
                        } else {
                            printf("Binary representation of %d is: ", number);
                            binary(number);
                            printf("\n");
                        }
                    
                        return 0;
                    }

Output

                    Binary representation of 10 is: 1010

                    Binary representation of 0 is: 0

Code 5


                    #include < stdio.h>

                    int largest(int arr[], int n) {
                        if (n == 1)
                            return arr[0];
                        int max = largest(arr, n - 1);
                        return (arr[n - 1] > max) ? arr[n - 1] : max;
                    }
                    
                    int main() {
                        int n;
                    
                        printf("Enter number of elements: ");
                        scanf("%d", &n);
                        
                        int arr[n];
                        printf("Enter elements:\n");
                        for (int i = 0; i < n; i++) {
                            scanf("%d", &arr[i]);
                        }
                    
                        printf("Largest element: %d\n", largest(arr, n));
                    
                        return 0;
                    }

Output


                    Enter number of elements: 5
                    Enter elements:
                    2 7 3 1 5
                    
                    Largest element: 7

Code 7


                    #include < stdio.h>

                    int sumOfDigits(int n) {
                        if (n == 0)
                            return 0;
                        return (n % 10) + sumOfDigits(n / 10);
                    }
                    
                    int main() {
                        int number;
                    
                        printf("Enter a number: ");
                        scanf("%d", &number);
                    
                        int sum = sumOfDigits(number);
                        printf("Sum of digits: %d\n", sum);
                    
                        return 0;
                    }

Output


            
                    Enter a number: 1234
                    Sum of digits: 10

Code 8


                    #include < stdio.h>

                    double calculateMean(int arr[], int n) {
                        if (n == 0)
                            return 0;
                        return (arr[n - 1] + (n - 1) * calculateMean(arr, n - 1)) / n;
                    }
                    
                    int main() {
                        int n;
                    
                        printf("Enter number of elements: ");
                        scanf("%d", &n);
                        
                        int arr[n];
                        printf("Enter elements:\n");
                        for (int i = 0; i < n; i++) {
                            scanf("%d", &arr[i]);
                        }
                    
                        double mean = calculateMean(arr, n);
                        printf("Arithmetic mean: %.2f\n", mean);
                    
                        return 0;
                    }

Output


                    Enter number of elements: 5
                    Enter elements:
                    10
                    20
                    30
                    40
                    50
                    Arithmetic mean: 30.00
